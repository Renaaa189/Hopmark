
El proyecto consta de una base de datos que debe ser para almacenar estudiantes, aquellos estudiantes asistiran a 3 diferentes carreras:

Medicina, en la UBA.

Ingeniería en sistemas, en la UTN. 

y Economía(Contador Público), en la UADE. 

El programa deberá ser capaz de asignar materias a los alumnos de distintas carreas y de
diferentes años, y deberá de tener registro de años anteriores con lógica, es decir, Un 
alumno de 3er año en la carrera de medicina no puede:

- Tener desaprobadas materias de años anteriores.

- Tener materias no correspondientes a la carrera.

- No tener registro de sus años anteriores.

- Tener materias de años próximos/anteriores, se supone que deberá tener materias de ese mismo año.

Y... un detalle mas, por cada nota de cada materia los alumnos se les dará un 
CONEJO(ya se, raro) la edad o tipo de conejo dependerá de la calificación de cada materia, si la nota de
la materia es 1, 2 o 3 el conejo asignado será un conejo adulto, si la calificación de la materia es de 4,5,6 o 7, el 
conejo asignado será uno joven y si el conejo asignado es de 8, 9 o 10 el conejo que se le asignara será un Conejo Bebé.

Este proyecto requerirá de una estructura donde habrán dos carpetas principales, la api(usaremos javaScript) y
el client(React) así que bueno, yo ya tengo casi todo del proyecto pero me gustaría volver a empezar, así que decime que 
carpetas spn necesarias.

// api/controllers/alumnoController.js
const { Estudiante, Estudiante_Carrera, Estudiante_Materia, Materia, sequelize } = require('../models');

async function generarAlumnos() {
  try {
    const nombres = ["Juan","María","Pedro","Lucía","Carlos","Ana","Miguel","Sofía","Jorge","Camila","Andrés","Valentina","Martín","Julieta","Agustín","Florencia","Nicolás","Bruno","Paula","Sebastián"];
    const apellidos = ["García","Rodríguez","López","Pérez","Martínez","Gómez","Fernández","Díaz","Torres","Ruiz","Castro","Vázquez","Domínguez","Sosa","Navarro","Suárez","Romero","Herrera","Benítez","Molina"];
    const calles = ["San Martín","Belgrano","Rivadavia","Mitre","Lavalle","Moreno","25 de Mayo","Urquiza","Güemes","Independencia","Sarmiento","Perón","Alsina","Córdoba","Santa Fe"];

    const alumnos = [];
    const relacionesCarreras = [];

    let id_carrera = 1; // 1 = Ingeniería
    let count_ing = 0, count_med = 0, count_eco = 0;

    for (let i = 1; i <= 2000; i++) {
      const nombre = nombres[Math.floor(Math.random() * nombres.length)];
      const apellido = apellidos[Math.floor(Math.random() * apellidos.length)];
      const direccion = `${calles[Math.floor(Math.random() * calles.length)]} ${Math.floor(Math.random() * 4000) + 1}`;
      const telefono = `11-${Math.floor(10000000 + Math.random() * 90000000)}`;
      const dni = (30000000 + i).toString();

      // asignar carrera
      if (count_ing < 700) {
        id_carrera = 1; // Ingeniería
        count_ing++;
      } else if (count_med < 700) {
        id_carrera = 2; // Medicina
        count_med++;
      } else {
        id_carrera = 3; // Economía
        count_eco++;
      }

      alumnos.push({ dni, nombre, apellido, direccion, telefono });
    }

    // Insertar estudiantes
    const estudiantesInsertados = await Estudiante.bulkCreate(alumnos, { returning: true });

    // Crear relaciones estudiante_carrera
    for (const est of estudiantesInsertados) {
      let id_carrera_asignada;

      if (est.id_estudiante <= 700) id_carrera_asignada = 1;
      else if (est.id_estudiante <= 1400) id_carrera_asignada = 2;
      else id_carrera_asignada = 3;

      const anio_actual = id_carrera_asignada === 2
        ? Math.floor(1 + Math.random() * 6) // Medicina (6 años)
        : Math.floor(1 + Math.random() * 5); // Ingeniería / Economía (5 años)

      relacionesCarreras.push({
        id_estudiante: est.id_estudiante,
        id_carrera: id_carrera_asignada,
        regularidad: 'Regular',
        anio_actual
      });
    }

    await Estudiante_Carrera.bulkCreate(relacionesCarreras);

    console.log("✅ Se insertaron 2000 alumnos y se asignaron carrera + año correctamente.");

  } catch (error) {
    console.error("❌ Error al insertar alumnos y carreras:", error);
  }
}

async function obtenerAlumnoPorDni(req, res) {
  try {
    const { dni } = req.params;
    const alumno = await Estudiante.findOne({ where: { dni } });

    if (!alumno) return res.status(404).json({ mensaje: 'Alumno no encontrado' });

    res.json({
      id: alumno.id_estudiante || alumno.id || null,
      dni: alumno.dni,
      nombre: alumno.nombre,
      apellido: alumno.apellido,
      direccion: alumno.direccion,
      telefono: alumno.telefono,
    });
  } catch (error) {
    console.error('❌ Error en obtenerAlumnoPorDni:', error);
    res.status(500).json({ error: 'Error interno del servidor' });
  }
}

async function obtenerAlumnoConMaterias(req, res) {
  try {
    const { dni } = req.params;

    const alumno = await Estudiante.findOne({
      where: { dni },
      include: [
        {
          model: Estudiante_Materia,
          as: 'est_materias',
          include: [
            {
              model: Materia,
              as: 'materia',
              attributes: ['id_materia', 'nombre', 'anio_cursada']
            }
          ]
        }
      ],
      order: [[{ model: Estudiante_Materia, as: 'est_materias' }, 'anio_cursada', 'ASC']]
    });

    if (!alumno) return res.status(404).json({ mensaje: 'Alumno no encontrado' });

    const materias = (alumno.est_materias || []).map(em => ({
      id_materia: em.id_materia,
      materia: em.materia ? em.materia.nombre : null,
      anio_cursada: em.anio_cursada,
      estado_regularidad: em.estado_regularidad,
      nota_parcial_1: em.nota_parcial_1,
      nota_parcial_2: em.nota_parcial_2,
      nota_final: em.nota_final,
      estado_materia: em.estado_materia
    }));

    res.json({
      alumno: {
        id: alumno.id_estudiante || alumno.id || null,
        dni: alumno.dni,
        nombre: alumno.nombre,
        apellido: alumno.apellido,
        direccion: alumno.direccion,
        telefono: alumno.telefono
      },
      materias
    });

  } catch (error) {
    console.error('❌ Error en obtenerAlumnoConMaterias:', error);
    res.status(500).json({ error: 'Error interno del servidor' });
  }
}

module.exports = { generarAlumnos, obtenerAlumnoPorDni, obtenerAlumnoConMaterias };

const { Conejo, TipoConejo, Estudiante_Materia } = require('../models');

async function asignarConejosDesdeNotas() {
  try {
    const materiasAlumno = await Estudiante_Materia.findAll({
      include: ['materia', 'estudiante']
    });

    for (const em of materiasAlumno) {
      const nota = em.nota_final || 0;
      let tipoNombre;

      // Tipos de conejo según nota
      if (nota >= 8) tipoNombre = 'Conejo Bebé';
      else if (nota >= 4) tipoNombre = 'Conejo Joven';
      else tipoNombre = 'Conejo Adulto';

      // Buscamos o creamos el tipo de conejo
      let tipo = await TipoConejo.findOne({ where: { nombre: tipoNombre } });
      if (!tipo) {
        tipo = await TipoConejo.create({ nombre: tipoNombre, descripcion: `Asignado a alumnos con nota ${nota}` });
      }

      // Insert individual en Conejo
      await Conejo.create({
        id_estudiante: em.id_estudiante,
        id_materia: em.id_materia,
        id_tipo: tipo.id_tipo,
        anio_cursada: em.anio_cursada,
        nota_origen: nota
      });
    }

    console.log('✅ Se asignaron los conejos por nota correctamente.');
  } catch (error) {
    console.error('❌ Error asignando conejos:', error);
  }
}

async function getConejosPorEstudiante(req, res) {
  try {
    const { id_estudiante } = req.params;
    const conejos = await Conejo.findAll({ where: { id_estudiante }, include: ['materia', 'tipo'] });
    res.json(conejos);
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Error al obtener conejos' });
  }
}

async function getResumenConejosPorEstudiante(req, res) {
  try {
    const { id_estudiante } = req.params;
    const conejos = await Conejo.findAll({ where: { id_estudiante }, include: ['tipo'] });
    const resumen = {};

    for (const c of conejos) {
      const tipo = c.tipo.nombre;
      resumen[tipo] = (resumen[tipo] || 0) + 1;
    }

    res.json(resumen);
  } catch (error) {
    console.error(error);
    res.status(500).json({ error: 'Error al obtener resumen' });
  }
}

const { Estudiante, Estudiante_Carrera, Materia, Estudiante_Materia } = require('../models');
const { Sequelize } = require('sequelize');

async function generarEstudianteMateria(req, res) {
  try {
    console.log("📚 Generando Estudiante_Materia...");

    const estudiantes = await Estudiante.findAll();
    const carreras = await Estudiante_Carrera.findAll();
    const materias = await Materia.findAll();

    if (!estudiantes.length || !carreras.length || !materias.length) {
      return res?.json?.({ error: "Faltan datos básicos" }) || null;
    }

    const materiasPorCarrera = materias.reduce((acc, m) => {
      acc[m.id_carrera] = acc[m.id_carrera] || [];
      acc[m.id_carrera].push(m);
      return acc;
    }, {});

    const asignaciones = [];

    // ---------------------------------------------------------------------------------------
    // 1. Cada alumno recibe materias segun su año y carrera
    // ---------------------------------------------------------------------------------------
    for (const est of estudiantes) {
      const car = carreras.find(c => c.id_estudiante === est.id_estudiante);
      if (!car) continue;

      const listaMaterias = materiasPorCarrera[car.id_carrera] || [];

      // materias de AÑOS <= año_actual
      const materiasValidas = listaMaterias.filter(
        m => m.anio_cursada <= car.anio_actual
      );

      if (materiasValidas.length === 0) continue;

      for (const mat of materiasValidas) {
        asignaciones.push({
          id_estudiante: est.id_estudiante,
          id_materia: mat.id_materia,
          anio_cursada: mat.anio_cursada,
          nota_parcial_1: Math.floor(1 + Math.random() * 10),
          nota_parcial_2: Math.floor(1 + Math.random() * 10),
          nota_final: Math.floor(1 + Math.random() * 10)
        });
      }
    }

    // ---------------------------------------------------------------------------------------
    // 2. Asegurar mínimo 23 alumnos por materia
    // ---------------------------------------------------------------------------------------
    const materiasAgrupadas = {};

    for (const as of asignaciones) {
      materiasAgrupadas[as.id_materia] = materiasAgrupadas[as.id_materia] || [];
      materiasAgrupadas[as.id_materia].push(as);
    }

    for (const mat of materias) {
      const lista = materiasAgrupadas[mat.id_materia] || [];

      if (lista.length < 23) {
        const faltan = 23 - lista.length;
        const randomEstudiantes = estudiantes
          .sort(() => Math.random() - 0.5)
          .slice(0, faltan);

        for (const est of randomEstudiantes) {
          asignaciones.push({
            id_estudiante: est.id_estudiante,
            id_materia: mat.id_materia,
            anio_cursada: mat.anio_cursada,
            nota_parcial_1: Math.floor(1 + Math.random() * 10),
            nota_parcial_2: Math.floor(1 + Math.random() * 10),
            nota_final: Math.floor(1 + Math.random() * 10)
          });
        }
      }
    }

    // ---------------------------------------------------------------------------------------
    // 3. Insert masivo evitando duplicados
    // ---------------------------------------------------------------------------------------
    const unique = {};
    const finalRows = [];

    for (const row of asignaciones) {
      const key = `${row.id_estudiante}-${row.id_materia}`;
      if (!unique[key]) {
        unique[key] = true;

        const promedio = (row.nota_parcial_1 + row.nota_parcial_2 + row.nota_final) / 3;

        finalRows.push({
          ...row,
          estado_regularidad: promedio >= 4 ? "Regular" : "Libre",
          estado_materia:
            promedio >= 7 ? "Aprobada" :
            promedio >= 4 ? "Reprobada" :
            "Pendiente"
        });
      }
    }

    await Estudiante_Materia.bulkCreate(finalRows, {
      ignoreDuplicates: true
    });

    console.log("📚 Estudiante_Materia generado correctamente.");

    return res?.json?.({ ok: true, insertados: finalRows.length }) || null;

  } catch (err) {
    console.error("❌ Error:", err);
    return res?.status?.(500)?.json?.({ error: "Error generando datos" }) || null;
  }
}

module.exports = {
  generarEstudianteMateria
};

const { Estudiante_Materia, Conejo, TipoConejo } = require('../models');

async function asignarConejosDesdeNotas() {
  const registros = await Estudiante_Materia.findAll({ raw: true });

  const tipos = await TipoConejo.findAll();
  const tiposMap = {};
  tipos.forEach(t => tiposMap[t.nombre] = t.id_tipo);

  const conejos = [];

  for (const r of registros) {
    const notas = [r.nota_parcial_1, r.nota_parcial_2, r.nota_final];

    for (const nota of notas) {
      if (nota === null || nota === undefined) continue;
      const n = Number(nota);
      if (Number.isNaN(n)) continue;

      let tipo;
      if (n <= 3) tipo = 'Viejo';
      else if (n <= 7) tipo = 'Nuevo';
      else tipo = 'Bebe';

      conejos.push({
        id_estudiante: r.id_estudiante,
        id_materia: r.id_materia,
        id_tipo: tiposMap[tipo],
        anio_cursada: r.anio_cursada,
        nota_origen: n
      });
    }
  }

  if (conejos.length > 0) {
    await Conejo.bulkCreate(conejos, { ignoreDuplicates: true });
    console.log(`🐇 Se generaron ${conejos.length} conejos correctamente.`);
  } else {
    console.log('No hay notas para generar conejos.');
  }
}

module.exports = asignarConejosDesdeNotas;
